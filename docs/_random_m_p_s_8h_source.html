124  void Sampler::initialize() {
125  engine_.seed(seed_);
126 
127  const auto toml = toml::parse("setting.toml");
128  dBeta_ = toml::find<double>(toml, "tDMRG", "dBeta");
129  NBeta_ = toml::find<int>(toml, "tDMRG", "NBeta");
130  ObserveInterval_ = toml::find<int>(toml, "Sampling", "ObserveInterval");
131 
132  if (toml.contains("UnitaryTransformation")) {
133  n_uni_ = toml::find<int>(toml, "UnitaryTransformation", "Steps");
134  } else {
135  n_uni_ = 0;
136  }
137 
138  int MaxM = toml::find<int>(toml, "MPS", "MaxM");
139  double tol = toml::find<double>(toml, "MPS", "tol");
140 
141  tevol_args_ = itensor::Args(
142  "MaxDim", MaxM,
143  "Cutoff", tol
144  );
145 
146  output_["seed"] = seed_;
147 
148  beta_.reserve(NBeta_ / ObserveInterval_ + 1);
149  for (int i = 0; i < NBeta_; i++) {
150  if (i % ObserveInterval_ == 0) {
151  beta_.push_back(i*dBeta_);
152  }
153  }
154  beta_.push_back(NBeta_*dBeta_);
155  output_["beta"] = beta_;
156 
157  output_["LowestEnergy"] = nullptr;
158 
159  std::string file_pre("sample_"), file_suf(".json"), seed_str;
160  seed_str = std::to_string(seed_);
161  filename_ = file_pre + seed_str + file_suf;
162 
163  count_ = 0;

166  template<typename T>
167  void Sampler::run(T& observer) {
168 
169  nlohmann::json sample;
170  auto start = std::chrono::system_clock::now();
171  itensor::MPS psi;
172  if (PossibleQNs_.size() > 0) {
173  psi = RandomPhaseState::RandomPhaseState(sites_, PossibleQNs_, engine_);
174  } else {
175  psi = RandomPhaseState::RandomPhaseState(sites_, engine_);
176  }
177 
178  double nrm = psi.normalize();
179  for (int i = 0; i < n_uni_; i++) {
180  for (auto&& x : uni_gates_) {
181  psi.position(x.first);
182  itensor::applyGate(x.second, psi, tevol_args_);
183  nrm *= psi.normalize();
184  }
185  }
186  double ene_for_norm = 0.0;
187  double norm_factor = 1.0;
188 
189  for (int i = 0; i < NBeta_; i++) {
190  if (i % ObserveInterval_ == 0) {
191  observer(psi, sample);
192  sample["Norm"].push_back(nrm);
193  sample["BondDim"].push_back(itensor::maxLinkDim(psi));
194  double ene_present = sample["Energy"].back();
195 
196  if (i*0.5*dBeta_*std::abs(ene_present - ene_for_norm) > 1.0) {
197  double ene_diff = ene_present - ene_for_norm;
198  for (size_t j = 1; j < sample["Norm"].size(); j++) {
199  double val = sample["Norm"].at(j);
200  sample["Norm"].at(j) = val*std::exp(0.5*static_cast<double>(output_["beta"].at(j))*ene_diff);
201  }
202  nrm *= std::exp(i*0.5*dBeta_*ene_diff);
203  norm_factor *= std::exp(0.5*dBeta_*ene_diff);
204  ene_for_norm = ene_present;
205  }
206  }
207 
208  for (auto&& x : gates_) {
209  psi.position(x.first);
210  itensor::applyGate(x.second, psi, tevol_args_);
211  nrm *= psi.normalize();
212  }
213  nrm *= norm_factor;
214 
215  }
216  observer(psi, sample);
217  sample["Norm"].push_back(nrm);
218  sample["BondDim"].push_back(itensor::maxLinkDim(psi));
219  double ene_present = sample["Energy"].back();
220  double ene_diff = ene_present - ene_for_norm;
221  for (size_t j = 1; j < sample["Norm"].size(); j++) {
222  double val = sample["Norm"].at(j);
223  sample["Norm"].at(j) = val*std::exp(0.5*static_cast<double>(output_["beta"].at(j))*ene_diff);
224  }
225 
226  if (output_["LowestEnergy"].is_null() or output_["LowestEnergy"] > ene_present) {
227  output_["LowestEnergy"] = ene_present;
228  }
229 
230  output_["Samples"].push_back(sample);
231  auto end = std::chrono::system_clock::now();
232  double elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(end - start).count();
233  count_++;
234  std::cout << "Sample " << count_ << ", Elapsed time:" << elapsed / 1000 << "s, Norm:" << sample["Norm"].back() << std::endl;
235  output_["ElapsedTime"].push_back(elapsed/1000);
236  std::ofstream out_file(filename_);
237  out_file << output_ << std::endl;
238  }
